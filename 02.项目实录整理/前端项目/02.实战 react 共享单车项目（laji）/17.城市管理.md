class City extends Component{
	
	state = {};

	params = {
		page: 1
	};

	componentDidMount(){
		this.requestList();
	};
	
	//默认请求接口数据
	requestList = ()=>{
		let _this = this;
		Axios.ajax({
			url: '/open_city',
			data: {
				params:{
					page:this.params.page
				}
			}
		}).then((res)=>{
				this.setState({
					list: res.result.item_List.map((item, index)=>{
						item.key = index;
						return item;
					}),
					pagination: Utills.pagination(res, (current)=>{
						_this.params.page = current;
						this.requestList();
					})
				})
			})


	};

	};

				dataIndex: 'name'
			},{
